#include <stdio.h> //подключение библиотеки ввода/вывода
#include <windows.h> //библиотека для работы кириллицей
#include<math.h> //библиотека математических действий
void main(){ // функция main
	SetConsoleCP(1251); // изменение кодировки на кириллицу
	SetConsoleOutputCP(1251); // изменение кодировки на кириллицу
	double C, F, K; // объявление переменных 
	printf("Введите температуру в цельсиях  "); // запрашиваем ввести температуру в цельсиях
	scanf("%lf", &C); //ввод температуры в цельсиях
	F = C * 9 / 5 + 32; // вычисление температуры по фаренгейту
	K = C + 273.15; // вычисление температуры в кельвинах
	printf("Температура в фаренгейтах равна %lf \n", F); //вывод температуры по фаренгейту
	printf("Температура в кельвинах равна %lf \n", K); //вывод температуры в кельвинах

}
